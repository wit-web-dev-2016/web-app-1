View 2b - Bookmarks

In a separate view, each bookmark category:

  • Title of the Bookmark List

Each of the bookmarks has

  • Title
  • Link

Solution - Part 1

New version of bookmarklist-store.js

models/bookmarklist-store.js

'use strict';

const _ = require('lodash');

const bookmarklistCollection = require('./bookmarklist-store.json').bookmarklistCollection;

const bookmarklistStore = {

  bookmarklistCollection: require('./bookmarklist-store.json').bookmarklistCollection,

  getBookmarkListCollection() {
    return this.bookmarklistCollection;
  },

  getBookmarklist(id) {
    return _.find(this.bookmarklistCollection, { id: id });
  },
};

module.exports = bookmarklistStore;

New route:

routes.js

...
router.get('/bookmarklist/:id', bookmarklist.index);
...

A new controller:

controllers/bookmarklist.js

'use strict';

const logger = require('../utils/logger');
const bookmarklistStore = require('../models/bookmarklist-store');

const bookmarklist = {
  index(request, response) {
    const bookmarklistId = request.params.id;
    logger.debug('Bookmark id = ', bookmarklistId);
    const viewData = {
      title: 'Bookmarks',
      bookmarklist: bookmarklistStore.getBookmarklist(bookmarklistId),
    };
    response.render('bookmarklist', viewData);
  },

};

module.exports = bookmarklist;

View 2b - Delete Bookmark

Each of the bookmarks has - Delete button

Pressing the link will open the bookmark in a new window. Pressing the delete button will remove the bookmark from the list.

Solution

routes.js

...
router.get('/bookmarklist/:id/deletebookmark/:bookmarkid', bookmarklist.deleteBookmark);
...

views/bookmarklist.hbs

Inside the table, add a new column:

...
    <thead>
    <tr>
      <th>Title</th>
      <th>Link</th>
      <th></th>
    </tr>
    </thead>
...
...
        <td>
          <a href="/bookmarklist/{{../bookmarklist.id}}/deletebookmark/{{id}}" class="ui tiny red button">Delete Bookmark</a>
        </td>
...

controllers/bookmarklist.js

...

  deleteBookmark(request, response) {
    const bookmarklistId = request.params.id;
    const bookmarkId = request.params.bookmarkid;
    logger.debug(`Deleting Bookamrk ${bookmarkId} from Bookamrklist ${bookmarklistId}`);
    bookmarklistStore.removeBookmark(bookmarklistId, bookmarkId);
    response.redirect('/bookmarklist/' + bookmarklistId);
  },

...

models/bookmarklist-store.js

...
  removeBookmark(id, bookmarkId) {
    const bookmarklist = this.getBookmarklist(id);
    _.remove(bookmarklist.bookmarks, { id: bookmarkId });
  },
...
